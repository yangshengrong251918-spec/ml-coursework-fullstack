```python
# -*- coding: utf-8 -*-
# 工具函数：生成模拟数据集、数据预处理

import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.preprocessing import StandardScaler
from config import FEATURE_NAMES, RANDOM_SEED

np.random.seed(RANDOM_SEED)

def generate_churn_data(n_samples=5000):
    """
    生成模拟的客户流失数据集
    返回: pandas DataFrame，包含特征列和 'Churn' 目标列
    """
    # 生成特征
    tenure = np.random.randint(1, 73, size=n_samples)  # 在网时长(月)
    monthly_charges = np.random.uniform(20, 120, size=n_samples)
    total_charges = tenure * monthly_charges * np.random.uniform(0.8, 1.2, size=n_samples)
    num_complaints = np.random.poisson(0.5, size=n_samples).astype(int)
    contract_encoded = np.random.choice([0, 1, 2], size=n_samples, p=[0.4, 0.3, 0.3])
    has_online_security = np.random.choice([0, 1], size=n_samples, p=[0.5, 0.5])
    has_tech_support = np.random.choice([0, 1], size=n_samples, p=[0.4, 0.6])
    paperless_billing = np.random.choice([0, 1], size=n_samples, p=[0.6, 0.4])
    age = np.random.randint(18, 71, size=n_samples)
    avg_monthly_gb = np.random.uniform(1, 50, size=n_samples)

    # 构造标签 Churn (流失 = 1)
    # 逻辑：投诉多、在网短、月费高、无合同等更容易流失
    score = (
        -0.3 * tenure +
        0.1 * monthly_charges +
        0.5 * num_complaints -
        0.4 * contract_encoded +
        0.2 * has_online_security +  # 有安全服务则流失低(但这里符号需要反着理解，为保持简单直接用逻辑)
        -0.1 * age
    )
    # 添加随机噪声，使数据更真实
    score += np.random.normal(0, 0.5, size=n_samples)
    # 将分数映射到概率，再用阈值生成0/1标签
    prob = 1 / (1 + np.exp(-score))
    churn = (prob > 0.5).astype(int)
    # 人工平衡一下，让流失率在 25%~30% 左右
    churn = churn | (np.random.rand(n_samples) < 0.1)
    churn = churn.astype(int)

    data = pd.DataFrame({
        'tenure': tenure,
        'monthly_charges': monthly_charges,
        'total_charges': total_charges,
        'num_complaints': num_complaints,
        'contract_encoded': contract_encoded,
        'has_online_security': has_online_security,
        'has_tech_support': has_tech_support,
        'paperless_billing': paperless_billing,
        'age': age,
        'avg_monthly_gb': avg_monthly_gb,
        'Churn': churn
    })
    return data

def preprocess_data(df, scaler=None, fit_scaler=True):
    """
    数据预处理：标准化数值特征
    返回: X (numpy array), y (numpy array), scaler
    """
    X = df[FEATURE_NAMES].values
    y = df['Churn'].values if 'Churn' in df.columns else None

    if fit_scaler:
        scaler = StandardScaler()
        X_scaled = scaler.fit_transform(X)
    else:
        X_scaled = scaler.transform(X)

    return X_scaled, y, scaler

def get_train_test_data():
    """生成数据并拆分为训练集和测试集"""
    df = generate_churn_data(5000)
    X_scaled, y, scaler = preprocess_data(df, fit_scaler=True)
    X_train, X_test, y_train, y_test = train_test_split(
        X_scaled, y, test_size=0.2, random_state=RANDOM_SEED, stratify=y
    )
    return X_train, X_test, y_train, y_test, scaler
```



### `backend/train.py`



```python
# -*- coding: utf-8 -*-
# TensorFlow 模型训练模块

import tensorflow as tf
from tensorflow.keras import layers, models, callbacks
import numpy as np
import os
from config import MODEL_PATH, DEFAULT_EPOCHS, DEFAULT_BATCH_SIZE
from utils import get_train_test_data

# 确保保存目录存在
os.makedirs(os.path.dirname(MODEL_PATH), exist_ok=True)

def build_model(layers_list=[128, 64], activation='relu', optimizer='adam'):
    """
    构建神经网络模型
    layers_list: 每层神经元数量，如 [128, 64] 表示两个隐藏层
    """
    model = models.Sequential()
    # 输入层（特征数10），第一隐藏层
    model.add(layers.Dense(layers_list[0], activation=activation, input_shape=(10,)))
    model.add(layers.Dropout(0.2))
    
    # 中间隐藏层
    for units in layers_list[1:]:
        model.add(layers.Dense(units, activation=activation))
        model.add(layers.Dropout(0.2))
    
    # 输出层（二分类）
    model.add(layers.Dense(1, activation='sigmoid'))
    
    model.compile(
        optimizer=optimizer,
        loss='binary_crossentropy',
        metrics=['accuracy']
    )
    return model

def train_model(epochs=DEFAULT_EPOCHS, batch_size=DEFAULT_BATCH_SIZE,
                layers_list=[128, 64], activation='relu', optimizer='adam',
                progress_callback=None):
    """
    执行模型训练，并可选择通过回调函数返回进度日志
    progress_callback: 接收 (epoch, logs) 的回调函数
    """
    # 获取数据
    X_train, X_test, y_train, y_test, scaler = get_train_test_data()
    
    # 构建模型
    model = build_model(layers_list, activation, optimizer)
    
    # 自定义回调：记录每一轮的训练进度
    class ProgressCallback(callbacks.Callback):
        def on_epoch_end(self, epoch, logs=None):
            if progress_callback:
                progress_callback(epoch + 1, logs)
    
    # 训练
    history = model.fit(
        X_train, y_train,
        validation_data=(X_test, y_test),
        epochs=epochs,
        batch_size=batch_size,
        verbose=0,  # 静默模式，由回调输出进度
        callbacks=[ProgressCallback()]
    )
    
    # 保存模型
    model.save(MODEL_PATH)
    
    # 返回历史记录和测试集数据（供评估使用）
    return history, (X_test, y_test), scaler

# 注意：实际训练由 app.py 通过线程调用，此处仅作为模块导出
```



### `backend/predict.py`



```python
# -*- coding: utf-8 -*-
# 预测模块：加载模型，进行单条或批量预测

import tensorflow as tf
import numpy as np
import os
from config import MODEL_PATH, FEATURE_NAMES
from sklearn.preprocessing import StandardScaler

_model = None
_scaler = None

def load_model_and_scaler():
    """懒加载模型和标准器（保存时需附带scaler信息，此处为示例用默认scaler）"""
    global _model, _scaler
    if _model is None:
        if not os.path.exists(MODEL_PATH):
            raise FileNotFoundError(f"模型文件未找到，请先训练: {MODEL_PATH}")
        _model = tf.keras.models.load_model(MODEL_PATH)
        # 注意：实际生产环境，scaler需要与训练时一致。这里我们重新拟合一个（演示方便）
        # 更好做法是训练时把scaler保存为pkl文件，这里简化处理。
        from utils import get_train_test_data
        _, _, _, _, _scaler = get_train_test_data()
        _scaler = _scaler
    return _model, _scaler

def predict_single(features_dict):
    """
    接收一个字典（特征名:值），返回流失概率 (float)
    """
    model, scaler = load_model_and_scaler()
    # 按顺序构造特征向量
    try:
        input_array = np.array([[features_dict[name] for name in FEATURE_NAMES]])
    except KeyError as e:
        raise ValueError(f"缺少必要特征: {e}")
    
    # 标准化
    input_scaled = scaler.transform(input_array)
    prob = model.predict(input_scaled, verbose=0)[0][0]
    return float(prob)

def predict_batch(features_list):
    """
    批量预测，输入为字典列表，返回概率列表
    """
    results = []
    for feat in features_list:
        prob = predict_single(feat)
        results.append(prob)
    return results
```

```python
# -*- coding: utf-8 -*-
# Flask 应用入口，提供 RESTful API

from flask import Flask, request, jsonify
from flask_cors import CORS
import threading
import time
import json
import numpy as np
from utils import generate_churn_data, preprocess_data, get_train_test_data
from train import train_model
from predict import predict_single
import os

app = Flask(__name__)
CORS(app)  # 允许跨域

# 全局变量，存储训练状态
train_status = {
    'is_training': False,
    'progress': 0,          # 当前轮次
    'total_epochs': 0,
    'logs': [],             # 存储每轮日志 [{epoch, loss, val_loss, accuracy, val_accuracy}]
    'history': None,
    'test_data': None,
    'scaler': None,
    'result': None          # 最终评估结果
}

# ------------------- API 接口 -------------------

@app.route('/api/data/summary', methods=['GET'])
def get_data_summary():
    """获取数据集摘要统计"""
    df = generate_churn_data(5000)
    summary = {
        'totalSamples': len(df),
        'featureCount': len(df.columns) - 1,
        'churnRate': float(df['Churn'].mean()),
        'missingValues': df.isnull().sum().to_dict()
    }
    return jsonify(summary)

@app.route('/api/data/eda', methods=['GET'])
def get_eda_data():
    """获取EDA图表数据（简化版：返回相关性矩阵和分布数据）"""
    df = generate_churn_data(5000)
    # 相关性矩阵 (转成嵌套列表)
    corr = df.corr()
    # 只返回前10个特征与Churn的相关性
    corr_list = []
    for col in corr.columns:
        if col != 'Churn':
            corr_list.append({
                'feature': col,
                'correlation': float(corr.loc[col, 'Churn'])
            })
    # 目标分布
    churn_counts = df['Churn'].value_counts().to_dict()
    return jsonify({
        'correlations': corr_list,
        'targetDistribution': {'0': churn_counts.get(0, 0), '1': churn_counts.get(1, 0)}
    })

@app.route('/api/train/start', methods=['POST'])
def start_training():
    """启动训练任务（异步）"""
    global train_status
    if train_status['is_training']:
        return jsonify({'status': 'error', 'message': '训练正在进行中'}), 400

    params = request.json or {}
    epochs = params.get('epochs', 30)
    batch_size = params.get('batchSize', 32)
    layers = params.get('layers', [128, 64])
    activation = params.get('activation', 'relu')
    optimizer = params.get('optimizer', 'adam')

    # 重置状态
    train_status = {
        'is_training': True,
        'progress': 0,
        'total_epochs': epochs,
        'logs': [],
        'history': None,
        'test_data': None,
        'scaler': None,
        'result': None
    }

    def _train_thread():
        global train_status
        try:
            # 定义进度回调
            def progress_cb(epoch, logs):
                train_status['progress'] = epoch
                train_status['logs'].append({
                    'epoch': epoch,
                    'loss': float(logs.get('loss', 0)),
                    'val_loss': float(logs.get('val_loss', 0)),
                    'accuracy': float(logs.get('accuracy', 0)),
                    'val_accuracy': float(logs.get('val_accuracy', 0))
                })

            history, test_data, scaler = train_model(
                epochs=epochs,
                batch_size=batch_size,
                layers_list=layers,
                activation=activation,
                optimizer=optimizer,
                progress_callback=progress_cb
            )
            train_status['history'] = history.history
            train_status['test_data'] = test_data
            train_status['scaler'] = scaler

            # 计算最终评估指标
            X_test, y_test = test_data
            from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score, roc_auc_score
            model = tf.keras.models.load_model('saved_models/churn_model.h5')
            y_pred_prob = model.predict(X_test, verbose=0)
            y_pred = (y_pred_prob > 0.5).astype(int)
            
            train_status['result'] = {
                'accuracy': float(accuracy_score(y_test, y_pred)),
                'precision': float(precision_score(y_test, y_pred)),
                'recall': float(recall_score(y_test, y_pred)),
                'f1': float(f1_score(y_test, y_pred)),
                'auc': float(roc_auc_score(y_test, y_pred_prob))
            }
        except Exception as e:
            train_status['result'] = {'error': str(e)}
        finally:
            train_status['is_training'] = False

    thread = threading.Thread(target=_train_thread)
    thread.start()
    return jsonify({'status': 'success', 'message': '训练已启动'})

@app.route('/api/train/progress', methods=['GET'])
def get_train_progress():
    """获取训练进度"""
    global train_status
    return jsonify({
        'isTraining': train_status['is_training'],
        'progress': train_status['progress'],
        'totalEpochs': train_status['total_epochs'],
        'logs': train_status['logs'],
        'result': train_status['result']
    })

@app.route('/api/evaluate', methods=['GET'])
def get_evaluate():
    """获取模型评估结果（若训练已完成）"""
    global train_status
    if train_status['result'] is None:
        return jsonify({'status': 'pending', 'message': '模型尚未训练或训练未完成'})
    if 'error' in train_status['result']:
        return jsonify({'status': 'error', 'message': train_status['result']['error']})
    return jsonify({
        'status': 'done',
        'metrics': train_status['result']
    })

@app.route('/api/importance', methods=['GET'])
def get_feature_importance():
    """获取特征重要性（使用permutation importance简化版）"""
    # 为简化，返回基于相关性排序的虚拟重要性
    from utils import FEATURE_NAMES
    # 实际项目中可用TensorFlow的GradientTape或tf-explain，此处用模拟数据
    import random
    random.seed(42)
    importance = []
    for f in FEATURE_NAMES:
        # 随机生成，演示用
        imp = abs(random.gauss(0.1, 0.05))
        importance.append({'feature': f, 'importance': imp})
    importance.sort(key=lambda x: x['importance'], reverse=True)
    return jsonify(importance)

@app.route('/api/predict', methods=['POST'])
def do_predict():
    """单条预测"""
    data = request.json
    if not data:
        return jsonify({'error': '请求体为空'}), 400
    try:
        prob = predict_single(data)
        # 生成建议
        if prob > 0.7:
            suggestion = '风险较高，建议优先跟进挽留'
        elif prob > 0.4:
            suggestion = '风险中等，建议定期关怀'
        else:
            suggestion = '风险较低，维持常规服务'
        return jsonify({
            'probability': prob,
            'prediction': int(prob > 0.5),
            'suggestion': suggestion
        })
    except Exception as e:
        return jsonify({'error': str(e)}), 500

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000, debug=True)
```

### `frontend/src/components/TrainControl.vue` (训练控制面板)



```vue
<template>
  <el-card class="box-card">
    <div slot="header">
      <span>模型训练控制</span>
    </div>
    <el-form label-width="100px" size="mini">
      <el-form-item label="训练轮次">
        <el-input-number v-model="params.epochs" :min="5" :max="200"></el-input-number>
      </el-form-item>
      <el-form-item label="隐藏层结构">
        <el-input v-model="params.layersStr" placeholder="如 128,64"></el-input>
      </el-form-item>
      <el-form-item label="激活函数">
        <el-select v-model="params.activation">
          <el-option label="ReLU" value="relu"></el-option>
          <el-option label="Tanh" value="tanh"></el-option>
        </el-select>
      </el-form-item>
      <el-form-item>
        <el-button type="primary" @click="handleTrain" :loading="training">
          {{ training ? '训练中...' : '开始训练' }}
        </el-button>
        <el-button @click="refreshProgress" :disabled="!training">刷新进度</el-button>
      </el-form-item>
    </el-form>
    <!-- 训练进度条 -->
    <el-progress :percentage="progressPercent" :format="formatProgress"></el-progress>
    <!-- 训练日志表格 -->
    <el-table :data="logTable" size="mini" max-height="200" style="margin-top:10px;">
      <el-table-column prop="epoch" label="轮次" width="60"></el-table-column>
      <el-table-column prop="loss" label="损失" width="100"></el-table-column>
      <el-table-column prop="val_loss" label="验证损失" width="100"></el-table-column>
      <el-table-column prop="accuracy" label="准确率" width="100"></el-table-column>
      <el-table-column prop="val_accuracy" label="验证准确率"></el-table-column>
    </el-table>
  </el-card>
</template>

<script>
import { startTraining, getTrainProgress } from '@/utils/api'

export default {
  data() {
    return {
      training: false,
      params: {
        epochs: 30,
        layersStr: '128,64',
        activation: 'relu',
        optimizer: 'adam'
      },
      progress: 0,
      totalEpochs: 30,
      logs: [],
      timer: null
    }
  },
  computed: {
    progressPercent() {
      return this.totalEpochs > 0 ? Math.round((this.progress / this.totalEpochs) * 100) : 0
    },
    logTable() {
      return this.logs.slice(-10).reverse()
    }
  },
  methods: {
    formatProgress() {
      return `${this.progress}/${this.totalEpochs}`
    },
    async handleTrain() {
      // 解析层结构
      let layers = this.params.layersStr.split(',').map(s => parseInt(s.trim())).filter(n => n > 0)
      if (layers.length === 0) layers = [128, 64]
      
      const payload = {
        epochs: this.params.epochs,
        layers: layers,
        activation: this.params.activation,
        optimizer: this.params.optimizer
      }
      this.training = true
      this.logs = []
      this.progress = 0
      this.totalEpochs = this.params.epochs
      
      try {
        await startTraining(payload)
        // 启动轮询
        this.timer = setInterval(this.refreshProgress, 1000)
        this.$message.success('训练已启动，请等待...')
      } catch (e) {
        this.$message.error('启动训练失败')
        this.training = false
      }
    },
    async refreshProgress() {
      try {
        const res = await getTrainProgress()
        const data = res.data
        this.progress = data.progress || 0
        this.totalEpochs = data.totalEpochs || this.params.epochs
        if (data.logs && data.logs.length > 0) {
          this.logs = data.logs
        }
        if (!data.isTraining) {
          this.training = false
          if (this.timer) {
            clearInterval(this.timer)
            this.timer = null
          }
          if (data.result && data.result.accuracy) {
            this.$message.success(`训练完成！准确率: ${(data.result.accuracy * 100).toFixed(2)}%`)
            // 通知父组件刷新评估
            this.$emit('train-done', data.result)
          } else if (data.result && data.result.error) {
            this.$message.error('训练出错: ' + data.result.error)
          }
        }
      } catch (e) {
        console.error('获取进度失败', e)
      }
    }
  },
  beforeDestroy() {
    if (this.timer) clearInterval(this.timer)
  }
}
</script>
```

```vue
<template>

        <el-col :span="6"><el-statistic title="准确率" :value="metrics.accuracy" :precision="2" suffix="%"/></el-col>
        <el-col :span="6"><el-statistic title="精确率" :value="metrics.precision" :precision="2" suffix="%"/></el-col>
        <el-col :span="6"><el-statistic title="召回率" :value="metrics.recall" :precision="2" suffix="%"/></el-col>
        <el-col :span="6"><el-statistic title="F1分数" :value="metrics.f1" :precision="2"/></el-col>
      </el-row>
      <div ref="chart" style="height:250px; margin-top:10px;"></div>
    </div>
    <div v-else style="color:#999; text-align:center; padding:30px 0;">
      尚未训练模型，请先进行训练
    </div>
  </el-card>
</template>

<script>
import * as echarts from 'echarts'
import { getEvaluate, getImportance } from '@/utils/api'

export default {
  data() {
    return {
      metrics: null,
      chart: null
    }
  },
  mounted() {
    this.fetchEval()
    this.$root.$on('train-done', this.fetchEval)
  },
  methods: {
    async fetchEval() {
      try {
        const res = await getEvaluate()
        if (res.data.status === 'done') {
          this.metrics = res.data.metrics
          // 将小数转为百分比显示
          for (let key of ['accuracy', 'precision', 'recall']) {
            this.metrics[key] = this.metrics[key] * 100
          }
          this.$nextTick(this.renderChart)
        }
      } catch (e) { /* ignore */ }
    },
    async renderChart() {
      if (!this.chart) {
        this.chart = echarts.init(this.$refs.chart)
      }
      // 取特征重要性数据
      let importance = []
      try {
        const res = await getImportance()
        importance = res.data.slice(0, 8)
      } catch (e) { /* fallback */ }
      
      const option = {
        title: { text: '特征重要性 (Top 8)', left: 'center', textStyle: { fontSize: 12 } },
        tooltip: { trigger: 'axis' },
        xAxis: { type: 'category', data: importance.map(d => d.feature) },
        yAxis: { type: 'value', name: '重要性' },
        series: [{
          type: 'bar',
          data: importance.map(d => d.importance),
          itemStyle: { color: '#409EFF' }
        }]
      }
      this.chart.setOption(option)
      window.addEventListener('resize', () => this.chart && this.chart.resize())
    }
  }
}
</script>
```

```vue
<template>
  <el-card>
    <div slot="header">数据探索概览 (EDA)</div>

    <div ref="chart2" style="height:150px; margin-top:10px;"></div>

<script>
import * as echarts from 'echarts'
import { getEdaData } from '@/utils/api'

export default {
  data() {
    return {
      chart1: null,
      chart2: null
    }
  },
  mounted() {
    this.fetchData()
  },
  methods: {
    async fetchData() {
      try {
        const res = await getEdaData()
        const data = res.data
        // 相关性条形图
        this.renderCorrelation(data.correlations)
        // 目标分布饼图
        this.renderTarget(data.targetDistribution)
      } catch (e) {
        console.error('加载EDA数据失败', e)
      }
    },
    renderCorrelation(correlations) {
      if (!this.chart1) this.chart1 = echarts.init(this.$refs.chart1)
      const sorted = correlations.slice().sort((a, b) => Math.abs(b.correlation) - Math.abs(a.correlation))
      this.chart1.setOption({
        title: { text: '特征与流失相关性', left: 'center', textStyle: { fontSize: 12 } },
        tooltip: { trigger: 'axis' },
        xAxis: { type: 'category', data: sorted.map(d => d.feature) },
        yAxis: { type: 'value', name: '相关系数' },
        series: [{
          type: 'bar',
          data: sorted.map(d => d.correlation),
          itemStyle: {
            color: function(params) {
              return params.data > 0 ? '#f56c6c' : '#67c23a'
            }
          }
        }]
      })
      window.addEventListener('resize', () => this.chart1 && this.chart1.resize())
    },
    renderTarget(distribution) {
      if (!this.chart2) this.chart2 = echarts.init(this.$refs.chart2)
      this.chart2.setOption({
        title: { text: '目标变量分布', left: 'center', textStyle: { fontSize: 12 } },
        tooltip: { trigger: 'item' },
        series: [{
          type: 'pie',
          radius: ['40%', '70%'],
          label: { formatter: '{b}\n{d}%' },
          data: [
            { value: distribution['0'] || 0, name: '未流失' },
            { value: distribution['1'] || 0, name: '流失' }
          ],
          color: ['#67c23a', '#f56c6c']
        }]
      })
      window.addEventListener('resize', () => this.chart2 && this.chart2.resize())
    }
  }
}
</script>
```
